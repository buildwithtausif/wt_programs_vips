inbuilt functions & dialog box

        
// Using dialog boxes
alert("Welcome! This program shows usage of JS inbuilt functions.");
// Prompt box
let name = prompt("Enter your name:");

// Confirm box
let isReady = confirm("Do you want to continue?");

if (isReady) {
    // Using some inbuilt functions

    // String functions
    let upperName = name.toUpperCase();
    alert("Your name in uppercase: " + upperName);

    // Math functions
    let num = 25.67;
    alert("Number: " + num);
    alert("Math.round(): " + Math.round(num));
    alert("Math.floor(): " + Math.floor(num));
    alert("Math.ceil(): " + Math.ceil(num));
    alert("Math.random() (0-1): " + Math.random());
    } else {
        alert("You chose to exit. Goodbye!");
    }
